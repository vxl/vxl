    include_directories( ${BRL_INCLUDE_DIR}/bpro/core )

      if(PYTHON_LIBRARIES)
        target_link_libraries( brec_batch optimized ${PYTHON_LIBRARY})
      endif(PYTHON_LIBRARIES)

      if( BUILD_TESTING )
        add_subdirectory(tests)
